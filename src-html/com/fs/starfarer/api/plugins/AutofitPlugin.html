021public interface AutofitPlugin {
022        public interface AvailableWeapon {
023                String getId();
024                WeaponSpecAPI getSpec();
025                float getPrice();
026                CargoAPI getSource();
027                SubmarketAPI getSubmarket();
028                int getQuantity();
029                void setQuantity(int quantity);
030                float getOPCost(MutableCharacterStatsAPI stats, MutableShipStatsAPI shipStats);
031        }
032        
033        public interface AvailableFighter {
034                String getId();
035                FighterWingSpecAPI getWingSpec();
036                float getPrice();
037                CargoAPI getSource();
038                SubmarketAPI getSubmarket();
039                int getQuantity();
040                void setQuantity(int quantity);
041        }
042        
043        public interface AutofitPluginDelegate {
044                void fitFighterInSlot(int index, AvailableFighter fighter, ShipVariantAPI variant);
045                void clearFighterSlot(int index, ShipVariantAPI variant);
046                void fitWeaponInSlot(WeaponSlotAPI slot, AvailableWeapon weapon, ShipVariantAPI variant);
047                void clearWeaponSlot(WeaponSlotAPI slot, ShipVariantAPI variant);
048                
049                List<AvailableWeapon> getAvailableWeapons();
050                List<AvailableFighter> getAvailableFighters();
051                
052                boolean isPriority(WeaponSpecAPI weapon);
053                boolean isPriority(FighterWingSpecAPI wing);
054                
055                List<String> getAvailableHullmods();
056                void syncUIWithVariant(ShipVariantAPI variant);
057                
058                //void syncUIWithVariant();
059                
060                ShipAPI getShip();
061                
062                FactionAPI getFaction();
063                
064                boolean isAllowSlightRandomization();
065                
066                boolean isPlayerCampaignRefit();
067                boolean canAddRemoveHullmodInPlayerCampaignRefit(String modId);
068        }
069        
070
071        
072        
073        public static class AutofitOption {
074                public String id;
075                public String text;
076                public boolean checked;
077                public String tooltip;
078                public AutofitOption(String id, String text, boolean checked, String tooltip) {
079                        this.id = id;
080                        this.text = text;
081                        this.checked = checked;
082                        this.tooltip = tooltip;
083                }
084        }
085        
086        
087        //void init(AutofitPluginDelegate delegate);
088        List<AutofitOption> getOptions();
089        void doFit(ShipVariantAPI current, ShipVariantAPI target, int maxSMods, AutofitPluginDelegate delegate);
090        float getRating(ShipVariantAPI current, ShipVariantAPI target, AutofitPluginDelegate delegate);
091        int getCreditCost();
092        
093
094        String getQuickActionText();
095        void doQuickAction(ShipVariantAPI current, AutofitPluginDelegate delegate);
096        String getQuickActionTooltip();
097        
098        
099        void autoAssignOfficers(CampaignFleetAPI fleet);
100        boolean isQuickActionEnabled(ShipVariantAPI currentVariant);
101        void setRandom(Random random);
102}
