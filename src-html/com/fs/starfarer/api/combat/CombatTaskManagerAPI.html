001package com.fs.starfarer.api.combat;
002
003import java.util.List;
004
005import com.fs.starfarer.api.combat.CombatFleetManagerAPI.AssignmentInfo;
006
007public interface CombatTaskManagerAPI {
008        /**
009         * Returns the current assignment for a ship (the assignment type, and the target, if any).
010         * Returns null if there isn't one (i.e. the ship is on a default search-and-destroy).
011         * 
012         * For fighter wings, can pass in any fighter from the wing to get the assignment.
013         * 
014         * @param ship
015         * @return
016         */
017        AssignmentInfo getAssignmentFor(ShipAPI ship);
018        List<AssignmentInfo> getAllAssignments();
019        
020        
021        /**
022         * target should be one of:
023         *      BattleObjectiveAPI
024         *      DeployedFleetMemberAPI
025         *      the result of createWaypoint()
026         * 
027         * @param type
028         * @param target
029         * @param useCommandPointIfNeeded
030         * @return
031         */
032        AssignmentInfo createAssignment(CombatAssignmentType type, AssignmentTargetAPI target, boolean useCommandPoint);
033        
034        void giveAssignment(DeployedFleetMemberAPI member, AssignmentInfo assignment, boolean useCommandPointIfNeeded);
035        void orderRetreat(DeployedFleetMemberAPI member, boolean useCommandPointIfNeeded, boolean direct);
036        void orderSearchAndDestroy(DeployedFleetMemberAPI member, boolean useCommandPointIfNeeded);
037        
038        /**
039         * Cancels all assignments. New assignments can still be created.
040         */
041        void orderSearchAndDestroy();
042        
043        /**
044         * Cancels all assignment and orders all ships to retreat. Can not be aborted.
045         */
046        void orderFullRetreat();
047        
048        boolean isInFullRetreat();
049        MutableStat getCommandPointsStat();
050        int getCommandPointsLeft();
051        boolean isPreventFullRetreat();
052        void setPreventFullRetreat(boolean preventFullRetreat);
053        
054        boolean isFullAssault();
055        void setFullAssault(boolean explicitSearchAndDestroy);
056        
057        float getSecondsUntilNextPoint();
058        float getCPRateMult();
059        float getCPInterval();
060        MutableStat getCPRateModifier();
061        void removeAssignment(AssignmentInfo info);
062        void clearEmptyWaypoints();
063}
