001package com.fs.starfarer.api.impl.hullmods;
002
003import com.fs.starfarer.api.combat.BaseHullMod;
004import com.fs.starfarer.api.combat.MutableShipStatsAPI;
005import com.fs.starfarer.api.combat.ShipAPI;
006import com.fs.starfarer.api.combat.ShipAPI.HullSize;
007import com.fs.starfarer.api.impl.campaign.ids.HullMods;
008import com.fs.starfarer.api.impl.campaign.ids.Stats;
009
010public class RecoveryShuttles extends BaseHullMod {
011
012        public static float CREW_LOSS_MULT = 0.25f;
013        
014        public static float SMOD_CREW_LOSS_MULT = 0.05f;
015        
016        
017        public void applyEffectsBeforeShipCreation(HullSize hullSize, MutableShipStatsAPI stats, String id) {
018                boolean sMod = isSMod(stats);
019                float mult = CREW_LOSS_MULT;
020                if (sMod) mult = SMOD_CREW_LOSS_MULT; 
021                stats.getDynamic().getStat(Stats.FIGHTER_CREW_LOSS_MULT).modifyMult(id, mult);
022        }
023                
024        public String getSModDescriptionParam(int index, HullSize hullSize) {
025                if (index == 0) return "" + (int) ((1f - SMOD_CREW_LOSS_MULT) * 100f) + "%";
026                return null;
027        }
028        public String getDescriptionParam(int index, HullSize hullSize) {
029                if (index == 0) return "" + (int) ((1f - CREW_LOSS_MULT) * 100f) + "%";
030                return null;
031        }
032        
033        public boolean isApplicableToShip(ShipAPI ship) {
034                if (ship.getVariant().hasHullMod(HullMods.AUTOMATED)) return false;
035                
036                //int bays = (int) ship.getMutableStats().getNumFighterBays().getBaseValue();
037                int bays = (int) ship.getMutableStats().getNumFighterBays().getModifiedValue();
038//              if (ship != null && ship.getVariant().getHullSpec().getBuiltInWings().size() >= bays) {
039//                      return false;
040//              }
041                return ship != null && bays > 0; 
042        }
043        
044        public String getUnapplicableReason(ShipAPI ship) {
045                if (ship != null && ship.getVariant().hasHullMod(HullMods.AUTOMATED)) {
046                        return "Can not be installed on automated ships";
047                }
048                return "Ship does not have fighter bays";
049        }
050}
