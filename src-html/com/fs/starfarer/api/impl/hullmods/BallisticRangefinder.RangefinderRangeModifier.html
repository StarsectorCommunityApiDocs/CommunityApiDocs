036        public static WeaponSize getLargestBallisticSlot(ShipAPI ship) {
037                if (ship == null) return null;
038                WeaponSize largest = null;
039                for (WeaponSlotAPI slot : ship.getHullSpec().getAllWeaponSlotsCopy()) {
040                        if (slot.isDecorative() ) continue;
041                        if (slot.getWeaponType() == WeaponType.BALLISTIC) {
042                                if (largest == null || largest.ordinal() < slot.getSlotSize().ordinal()) {
043                                        largest = slot.getSlotSize();
044                                }
045                        }
046                }
047                return largest;
048        }

050        @Override
051        public void applyEffectsAfterShipCreation(ShipAPI ship, String id) {
052                WeaponSize largest = getLargestBallisticSlot(ship);
053                if (largest == null) return;
054                float small = 0f;
055                float medium = 0f;
056                float max = 0f;
057                switch (largest) {
058                case LARGE:
059                        small = BONUS_SMALL_3;
060                        medium = BONUS_MEDIUM_3;
061                        max = BONUS_MAX_3;
062                        break;
063                case MEDIUM:
064                        small = BONUS_SMALL_2;
065                        max = BONUS_MAX_2;
066                        break;
067                case SMALL:
068                        small = BONUS_SMALL_1;
069                        max = BONUS_MAX_1;
070                        break;
071                }
072                
073                ship.addListener(new RangefinderRangeModifier(small, medium, max));
074        }

077                public float small, medium, max;

090                public float getWeaponBaseRangeFlatMod(ShipAPI ship, WeaponAPI weapon) {
091//                      if (weapon.getSlot() == null || weapon.getSlot().getWeaponType() != WeaponType.BALLISTIC) {
092//                              return 0f;
093//                      }
094                        if (weapon.getSpec() == null) {
095                                return 0f;
096                        }
097                        if (weapon.getSpec().getMountType() != WeaponType.BALLISTIC && 
098                                        weapon.getSpec().getMountType() != WeaponType.HYBRID) {
099                                return 0f;
100                        }
101                        if (weapon.hasAIHint(AIHints.PD)) {
102                                return 0f;
103                        }
104                        
105                        float bonus = 0;
106                        if (weapon.getSize() == WeaponSize.SMALL) {
107                                bonus = small;
108                        } else if (weapon.getSize() == WeaponSize.MEDIUM) {
109                                bonus = medium;
110                        }
111                        if (weapon.getSpec().getMountType() == WeaponType.HYBRID) {
112                                bonus *= HYBRID_MULT;
113                                if (bonus < HYBRID_BONUS_MIN) {
114                                        bonus = HYBRID_BONUS_MIN;
115                                }
116                        }
117                        if (bonus == 0f) return 0f;
118                        
119                        float base = weapon.getSpec().getMaxRange();
120                        if (base + bonus > max) {
121                                bonus = max - base;
122                        }
123                        if (bonus < 0) bonus = 0;
124                        return bonus;
125                }

292        @Override
293        public boolean isApplicableToShip(ShipAPI ship) {
294                WeaponSize largest = getLargestBallisticSlot(ship);
295                if (ship != null && largest == null) {
296                        return false;
297                }
298                return getUnapplicableReason(ship) == null;
299        }
300        
301        public String getUnapplicableReason(ShipAPI ship) {
302                WeaponSize largest = getLargestBallisticSlot(ship);
303                if (ship != null && largest == null) {
304                        return "Ship has no ballistic weapon slots";
305                }
306                if (ship != null && 
307                                ship.getHullSize() != HullSize.CAPITAL_SHIP && 
308                                ship.getHullSize() != HullSize.DESTROYER && 
309                                ship.getHullSize() != HullSize.CRUISER) {
310                        return "Can only be installed on destroyer-class hulls and larger";
311                }
312                return null;
313        }
