001package com.fs.starfarer.api.impl.hullmods;
002
003import com.fs.starfarer.api.combat.BaseHullMod;
004import com.fs.starfarer.api.combat.MutableShipStatsAPI;
005import com.fs.starfarer.api.combat.ShipAPI;
006import com.fs.starfarer.api.combat.ShipAPI.HullSize;
007import com.fs.starfarer.api.impl.campaign.ids.Stats;
008
009public class DefectiveManufactory extends BaseHullMod {
010
011        //public static float SPEED_REDUCTION = 0.33333333f;
012        //public static float DAMAGE_INCREASE = 0.5f;
013        public static float SPEED_REDUCTION = 0.25f;
014        public static float DAMAGE_INCREASE = 0.25f;
015        
016        
017        public void applyEffectsToFighterSpawnedByShip(ShipAPI fighter, ShipAPI ship, String id) {
018                float effect = ship.getMutableStats().getDynamic().getValue(Stats.DMOD_EFFECT_MULT);
019                
020                MutableShipStatsAPI stats = fighter.getMutableStats();
021                
022                stats.getMaxSpeed().modifyMult(id, 1f - SPEED_REDUCTION * effect);
023                
024                stats.getArmorDamageTakenMult().modifyPercent(id, DAMAGE_INCREASE * 100f * effect);
025                stats.getShieldDamageTakenMult().modifyPercent(id, DAMAGE_INCREASE * 100f * effect);
026                stats.getHullDamageTakenMult().modifyPercent(id, DAMAGE_INCREASE * 100f * effect);
027                
028                //fighter.setHeavyDHullOverlay();
029                fighter.setLightDHullOverlay();
030        }
031        
032        public void applyEffectsBeforeShipCreation(HullSize hullSize, MutableShipStatsAPI stats, String id) {
033                CompromisedStructure.modifyCost(hullSize, stats, id);
034        }
035        
036                
037        public String getDescriptionParam(int index, HullSize hullSize, ShipAPI ship) {
038                float effect = 1f;
039                if (ship != null) effect = ship.getMutableStats().getDynamic().getValue(Stats.DMOD_EFFECT_MULT);
040                
041                if (index == 0) return "" + (int) Math.round(SPEED_REDUCTION * 100f * effect) + "%";
042                if (index == 1) return "" + (int) Math.round(DAMAGE_INCREASE * 100f * effect) + "%";
043                if (index >= 2) return CompromisedStructure.getCostDescParam(index, 2); 
044                return null;
045        }
046}
