001package com.fs.starfarer.api.impl.combat;
002
003import java.awt.Color;
004import java.util.List;
005
006import org.lwjgl.util.vector.Vector2f;
007
008import com.fs.starfarer.api.Global;
009import com.fs.starfarer.api.combat.BaseEveryFrameCombatPlugin;
010import com.fs.starfarer.api.combat.CombatEngineAPI;
011import com.fs.starfarer.api.combat.MissileAPI;
012import com.fs.starfarer.api.input.InputEventAPI;
013import com.fs.starfarer.api.util.IntervalUtil;
014import com.fs.starfarer.api.util.Misc;
015
016public class RiftTrailEffect extends BaseEveryFrameCombatPlugin {
017
018        protected IntervalUtil interval = new IntervalUtil(0.1f, 0.3f);
019        
020        protected MissileAPI missile;
021        protected String loopId;
022        
023        
024        public RiftTrailEffect(MissileAPI missile, String loopId) {
025                this.missile = missile;
026                this.loopId = loopId;
027        }
028
029        @Override
030        public void advance(float amount, List<InputEventAPI> events) {
031                if (Global.getCombatEngine().isPaused()) return;
032                
033                if (loopId != null) {
034                        Global.getSoundPlayer().playLoop(loopId, missile, 1f, missile.getBrightness(), 
035                                        missile.getLocation(), missile.getVelocity());
036                }
037                
038                interval.advance(amount);
039                if (interval.intervalElapsed()) {
040                        addParticles();
041                }
042                
043                if (missile.isExpired() || missile.didDamage() || !Global.getCombatEngine().isEntityInPlay(missile)) {
044                        Global.getCombatEngine().removePlugin(this);
045                }
046        }
047        
048
049        public void addParticles() {
050                CombatEngineAPI engine = Global.getCombatEngine();
051                Color c = RiftLanceEffect.getColorForDarkening(RiftCascadeEffect.STANDARD_RIFT_COLOR);
052                // subtracting the standard color looks better, makes the red a bit purplish
053                // inverting red to substract doesn't look as good for the trails
054//              MissileSpecAPI spec = missile.getSpec();
055//              c = spec.getExplosionColor();
056                
057                Color undercolor = RiftCascadeEffect.EXPLOSION_UNDERCOLOR;
058                
059                float b = missile.getCurrentBaseAlpha();
060                c = Misc.scaleAlpha(c, b);
061                undercolor = Misc.scaleAlpha(undercolor, b);
062                
063                float baseDuration = 4f;
064                float size = 30f;
065                size = missile.getSpec().getGlowRadius() * 0.5f;
066                
067                Vector2f point = new Vector2f(missile.getLocation());
068                Vector2f pointOffset = new Vector2f(missile.getVelocity());
069                pointOffset.scale(0.1f);
070                Vector2f.add(point, pointOffset, point);
071                
072                Vector2f vel = new Vector2f();
073                
074                for (int i = 0; i < 1; i++) {
075                        float dur = baseDuration + baseDuration * (float) Math.random();
076                        //float nSize = size * (1f + 0.0f * (float) Math.random());
077                        //float nSize = size * (0.75f + 0.5f * (float) Math.random());
078                        float nSize = size;
079                        Vector2f pt = Misc.getPointWithinRadius(point, nSize * 0.5f);
080                        Vector2f v = Misc.getUnitVectorAtDegreeAngle((float) Math.random() * 360f);
081                        v.scale(nSize + nSize * (float) Math.random() * 0.5f);
082                        v.scale(0.2f);
083                        Vector2f.add(vel, v, v);
084                        
085                        float maxSpeed = nSize * 1.5f * 0.2f; 
086                        float minSpeed = nSize * 1f * 0.2f; 
087                        float overMin = v.length() - minSpeed;
088                        if (overMin > 0) {
089                                float durMult = 1f - overMin / (maxSpeed - minSpeed);
090                                if (durMult < 0.1f) durMult = 0.1f;
091                                dur *= 0.5f + 0.5f * durMult;
092                        }
093                        engine.addNegativeNebulaParticle(pt, v, nSize * 1f, 2f,
094                                                                                        0.5f, 0f, dur, c);
095                }
096                
097                float dur = baseDuration; 
098                float rampUp = 0f;
099                rampUp = 0.5f;
100                c = undercolor;
101                for (int i = 0; i < 2; i++) {
102                        Vector2f loc = new Vector2f(point);
103                        loc = Misc.getPointWithinRadius(loc, size * 1f);
104                        float s = size * 3f * (0.5f + (float) Math.random() * 0.5f);
105                        engine.addNebulaParticle(loc, vel, s, 1.5f, rampUp, 0f, dur, c);
106                }
107        }
108        
109}
