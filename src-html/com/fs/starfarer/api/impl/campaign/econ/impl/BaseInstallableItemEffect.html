026public abstract class BaseInstallableItemEffect implements InstallableItemEffect {
027        public String id;
028        public SpecialItemSpecAPI spec;
029        public BaseInstallableItemEffect(String id) {
030                this.id = id;
031                spec = Global.getSettings().getSpecialItemSpec(id);
032        }

034        public float getItemDescPad(InstallableItemDescriptionMode mode) {
035                float pad = 0f;
036                if (mode == InstallableItemDescriptionMode.INDUSTRY_MENU_TOOLTIP ||
037                                mode == InstallableItemDescriptionMode.CARGO_TOOLTIP) {
038                        pad = 10f;
039                }
040                return pad;
041        }
042        public String getItemInstalledText(TooltipMakerAPI text, SpecialItemData data, InstallableItemDescriptionMode mode) {
043                String name = Misc.ucFirst(spec.getName().toLowerCase());
044                String pre = "";
045                float pad = 0f;
046                if (mode == InstallableItemDescriptionMode.MANAGE_ITEM_DIALOG_LIST ||
047                                mode == InstallableItemDescriptionMode.INDUSTRY_TOOLTIP) {
048                        pre = name + ". ";
049                } else if (mode == InstallableItemDescriptionMode.MANAGE_ITEM_DIALOG_INSTALLED || 
050                                mode == InstallableItemDescriptionMode.INDUSTRY_MENU_TOOLTIP) {
051                        pre = name + " currently installed. ";
052                }
053                if (mode == InstallableItemDescriptionMode.INDUSTRY_MENU_TOOLTIP ||
054                                mode == InstallableItemDescriptionMode.CARGO_TOOLTIP) {
055                        pad = 10f;
056                }
057                return pre;
058        }
059        
060        public void addItemDescription(Industry industry, TooltipMakerAPI text, SpecialItemData data, InstallableItemDescriptionMode mode) {
061                String pre = getItemInstalledText(text, data, mode);
062                float pad = getItemDescPad(mode);
063                addItemDescriptionImpl(industry, text, data, mode, pre, pad);
064        }

082        public List<String> getUnmetRequirements(Industry industry) {
083                return getUnmetRequirements(industry, false);
084        }
085        public List<String> getUnmetRequirements(Industry industry, boolean checkSurveyed) {
086                List<String> unmet = new ArrayList<String>();
087                if (industry == null) return unmet;
088                
089                MarketAPI market = industry.getMarket();
090                
091                boolean prelim = market.getSurveyLevel().ordinal() >= SurveyLevel.PRELIMINARY.ordinal();
092                boolean full = market.getSurveyLevel().ordinal() >= SurveyLevel.FULL.ordinal();
093                if (!checkSurveyed) {
094                        prelim = true;
095                        full = true;
096                }
097                
098                for (String curr : getRequirements(industry)) {
099                        if (ItemEffectsRepo.NO_ATMOSPHERE.equals(curr)) {
100                                if (!market.hasCondition(Conditions.NO_ATMOSPHERE) || !prelim) {
101                                        unmet.add(curr);
102                                }
103                        } else if (ItemEffectsRepo.HABITABLE.equals(curr)) {
104                                if (!market.hasCondition(Conditions.HABITABLE) || !prelim) {
105                                        unmet.add(curr);
106                                }
107                        } else if (ItemEffectsRepo.NOT_HABITABLE.equals(curr)) {
108                                if (market.hasCondition(Conditions.HABITABLE) || !prelim) {
109                                        unmet.add(curr);
110                                }
111                        } else if (ItemEffectsRepo.GAS_GIANT.equals(curr)) {
112                                if (market.getPlanetEntity() != null && !market.getPlanetEntity().isGasGiant()) {
113                                        unmet.add(curr);
114                                }
115                        } else if (ItemEffectsRepo.NOT_A_GAS_GIANT.equals(curr)) {
116                                if (market.getPlanetEntity() != null && market.getPlanetEntity().isGasGiant()) {
117                                        unmet.add(curr);
118                                }
119                        } else if (ItemEffectsRepo.NOT_EXTREME_WEATHER.equals(curr)) {
120                                if (market.hasCondition(Conditions.EXTREME_WEATHER) || !prelim) {
121                                        unmet.add(curr);
122                                }
123                        } else if (ItemEffectsRepo.NOT_EXTREME_TECTONIC_ACTIVITY.equals(curr)) {
124                                if (market.hasCondition(Conditions.EXTREME_TECTONIC_ACTIVITY) || !prelim) {
125                                        unmet.add(curr);
126                                }
127                        } else if (ItemEffectsRepo.NO_TRANSPLUTONIC_ORE_DEPOSITS.equals(curr)) {
128                                if ((market.hasCondition(Conditions.RARE_ORE_SPARSE) ||
129                                                market.hasCondition(Conditions.RARE_ORE_MODERATE) ||
130                                                market.hasCondition(Conditions.RARE_ORE_ABUNDANT) ||
131                                                market.hasCondition(Conditions.RARE_ORE_RICH) ||
132                                                market.hasCondition(Conditions.RARE_ORE_ULTRARICH)) || !full) {
133                                        unmet.add(curr);
134                                }
135                        } else if (ItemEffectsRepo.NO_VOLATILES_DEPOSITS.equals(curr)) {
136                                if ((market.hasCondition(Conditions.VOLATILES_TRACE) ||
137                                                market.hasCondition(Conditions.VOLATILES_DIFFUSE) ||
138                                                market.hasCondition(Conditions.VOLATILES_ABUNDANT) ||
139                                                market.hasCondition(Conditions.VOLATILES_PLENTIFUL)) || !full) {
140                                        unmet.add(curr);
141                                }
142                        } else if (ItemEffectsRepo.HOT_OR_EXTREME_HEAT.equals(curr)) {
143                                if ((!market.hasCondition(Conditions.HOT) &&
144                                                !market.hasCondition(Conditions.VERY_HOT)) || !prelim) {
145                                        unmet.add(curr);
146                                }
147                        } else if (ItemEffectsRepo.COLD_OR_EXTREME_COLD.equals(curr)) {
148                                if ((!market.hasCondition(Conditions.COLD) &&
149                                                !market.hasCondition(Conditions.VERY_COLD)) || !prelim) {
150                                        unmet.add(curr);
151                                }
152                        } else if (ItemEffectsRepo.CORONAL_TAP_RANGE.equals(curr)) {
153                                Pair<SectorEntityToken, Float> p = PopulationAndInfrastructure.getNearestCoronalTap(
154                                                                                                                market.getLocationInHyperspace(), true);
155                                float dist = Float.MAX_VALUE;
156                                if (p != null) dist = p.two;
157                                if (dist > ItemEffectsRepo.CORONAL_TAP_LIGHT_YEARS) {
158                                        unmet.add(curr);
159                                }
160                        }
161                }
162                return unmet;
163        }

165        @Override
166        public Set<String> getConditionsRelatedToRequirements(Industry industry) {
167                Set<String> cond = new LinkedHashSet<>();
168                
169                for (String curr : getRequirements(industry)) {
170                        if (ItemEffectsRepo.NO_ATMOSPHERE.equals(curr)) {
171                                cond.add(Conditions.NO_ATMOSPHERE);
172                        } else if (ItemEffectsRepo.HABITABLE.equals(curr)) {
173                                cond.add(Conditions.HABITABLE);
174                        } else if (ItemEffectsRepo.NOT_HABITABLE.equals(curr)) {
175                                cond.add(Conditions.HABITABLE);
176                        } else if (ItemEffectsRepo.NOT_EXTREME_WEATHER.equals(curr)) {
177                                cond.add(Conditions.EXTREME_WEATHER);
178                        } else if (ItemEffectsRepo.NOT_EXTREME_TECTONIC_ACTIVITY.equals(curr)) {
179                                cond.add(Conditions.EXTREME_TECTONIC_ACTIVITY);
180                        } else if (ItemEffectsRepo.NO_TRANSPLUTONIC_ORE_DEPOSITS.equals(curr)) {
181                                cond.add(Conditions.RARE_ORE_SPARSE);
182                                cond.add(Conditions.RARE_ORE_MODERATE);
183                                cond.add(Conditions.RARE_ORE_ABUNDANT);
184                                cond.add(Conditions.RARE_ORE_RICH);
185                                cond.add(Conditions.RARE_ORE_ULTRARICH);
186                        } else if (ItemEffectsRepo.NO_VOLATILES_DEPOSITS.equals(curr)) {
187                                cond.add(Conditions.VOLATILES_TRACE);
188                                cond.add(Conditions.VOLATILES_DIFFUSE);
189                                cond.add(Conditions.VOLATILES_ABUNDANT);
190                                cond.add(Conditions.VOLATILES_PLENTIFUL);
191                        } else if (ItemEffectsRepo.HOT_OR_EXTREME_HEAT.equals(curr)) {
192                                cond.add(Conditions.HOT);
193                                cond.add(Conditions.VERY_HOT);
194                        } else if (ItemEffectsRepo.COLD_OR_EXTREME_COLD.equals(curr)) {
195                                cond.add(Conditions.COLD);
196                                cond.add(Conditions.VERY_COLD);
197                        }
198                }
199                return cond;
200        }

204        protected void addRequirements(TooltipMakerAPI text, boolean canInstall, 
205                                                                InstallableItemDescriptionMode mode, String ... reqs) {
206                if (canInstall) return;
207                
208                float opad = 10f;
209                String list = "";
210                for (String curr : reqs) {
211                        curr = curr.trim();
212                        list += curr + ", ";
213                }
214                
215                Color reqColor = Misc.getBasePlayerColor();
216                if (mode != InstallableItemDescriptionMode.CARGO_TOOLTIP) {
217                        reqColor = Misc.getNegativeHighlightColor();
218                        opad = 0f;
219                }
220                
221                if (!list.isEmpty()) list = list.substring(0, list.length() - 2);
222                if (!list.isEmpty()) {
223                        text.addPara("Requires: " + list, opad, 
224                                        Misc.getGrayColor(), reqColor, reqs);
225                }
226        }
