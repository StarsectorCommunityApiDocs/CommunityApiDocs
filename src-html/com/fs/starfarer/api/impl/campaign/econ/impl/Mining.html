001package com.fs.starfarer.api.impl.campaign.econ.impl;
002
003import java.awt.Color;
004
005import com.fs.starfarer.api.Global;
006import com.fs.starfarer.api.campaign.PlanetAPI;
007import com.fs.starfarer.api.campaign.SpecialItemData;
008import com.fs.starfarer.api.campaign.econ.MarketAPI;
009import com.fs.starfarer.api.campaign.econ.MarketConditionAPI;
010import com.fs.starfarer.api.campaign.econ.MarketImmigrationModifier;
011import com.fs.starfarer.api.impl.campaign.econ.ResourceDepositsCondition;
012import com.fs.starfarer.api.impl.campaign.ids.Commodities;
013import com.fs.starfarer.api.impl.campaign.ids.Items;
014import com.fs.starfarer.api.impl.campaign.population.PopulationComposition;
015import com.fs.starfarer.api.ui.TooltipMakerAPI;
016import com.fs.starfarer.api.util.Misc;
017import com.fs.starfarer.api.util.Pair;
018
019
020public class Mining extends BaseIndustry implements MarketImmigrationModifier{
021
022        public void apply() {
023                super.apply(true);
024                
025                int size = market.getSize();
026                
027                demand(Commodities.HEAVY_MACHINERY, size - 3);
028                demand(Commodities.DRUGS, size);
029                
030                Pair<String, Integer> deficit = getMaxDeficit(Commodities.HEAVY_MACHINERY);
031                applyDeficitToProduction(0, deficit, 
032                                                Commodities.ORE,
033                                                Commodities.RARE_ORE,
034                                                Commodities.ORGANICS, 
035                                                Commodities.VOLATILES);
036                
037                if (!isFunctional()) {
038                        supply.clear();
039                }
040        }
041
042        
043        @Override
044        public void unapply() {
045                super.unapply();
046        }
047
048        protected boolean hasPostDemandSection(boolean hasDemand, IndustryTooltipMode mode) {
049                Pair<String, Integer> deficit = getMaxDeficit(Commodities.DRUGS);
050                if (deficit.two <= 0) return false;
051                //return mode == IndustryTooltipMode.NORMAL && isFunctional();
052                return mode != IndustryTooltipMode.NORMAL || isFunctional();
053        }
054        
055        @Override
056        protected void addPostDemandSection(TooltipMakerAPI tooltip, boolean hasDemand, IndustryTooltipMode mode) {
057                //if (mode == IndustryTooltipMode.NORMAL && isFunctional()) {
058                if (mode != IndustryTooltipMode.NORMAL || isFunctional()) {                     
059                        Color h = Misc.getHighlightColor();
060                        float opad = 10f;
061                        float pad = 3f;
062                        
063                        Pair<String, Integer> deficit = getMaxDeficit(Commodities.DRUGS);
064                        if (deficit.two > 0) {
065                                tooltip.addPara(getDeficitText(Commodities.DRUGS) + ": %s units. Reduced colony growth.", pad, h, "" + deficit.two);
066                        }
067                }
068        }
069        
070
071        @Override
072        public boolean isAvailableToBuild() {
073                if (!super.isAvailableToBuild()) return false;
074                
075                for (MarketConditionAPI mc : market.getConditions()) {
076                        String commodity = ResourceDepositsCondition.COMMODITY.get(mc.getId());
077                        if (commodity != null) {
078                                String industry = ResourceDepositsCondition.INDUSTRY.get(commodity);
079                                if (getId().equals(industry)) return true;
080                        }
081                }
082                return false;
083        }
084
085        @Override
086        public String getUnavailableReason() {
087                if (!super.isAvailableToBuild()) return super.getUnavailableReason();
088                
089                return "Requires resource deposits";
090        }
091
092        public void modifyIncoming(MarketAPI market, PopulationComposition incoming) {
093                Pair<String, Integer> deficit = getMaxDeficit(Commodities.DRUGS);
094                if (deficit.two > 0) {
095                        incoming.getWeight().modifyFlat(getModId(), -deficit.two, "Mining: drug shortage");
096                }
097        }
098        
099        @Override
100        public String getCurrentImage() {
101                float size = market.getSize();
102                if (market.getPlanetEntity() != null && market.getPlanetEntity().isGasGiant()) {
103                        return Global.getSettings().getSpriteName("industry", "mining_gas_giant");
104                }
105                if (size <= SIZE_FOR_SMALL_IMAGE) {
106                        return Global.getSettings().getSpriteName("industry", "mining_low");
107                }
108                return super.getCurrentImage();
109        }
110        
111        public float getPatherInterest() {
112                return 1f + super.getPatherInterest();
113        }
114        
115        @Override
116        protected boolean canImproveToIncreaseProduction() {
117                return true;
118        }
119        
120        
121        
122        public void applyVisuals(PlanetAPI planet) {
123                if (planet == null) return;
124                planet.getSpec().setShieldTexture2(Global.getSettings().getSpriteName("industry", "plasma_net_texture"));
125                planet.getSpec().setShieldThickness2(0.15f);
126                //planet.getSpec().setShieldColor2(new Color(255,255,255,175));
127                planet.getSpec().setShieldColor2(new Color(255,255,255,255));
128                planet.applySpecChanges();
129                shownPlasmaNetVisuals = true;
130        }
131        
132        public void unapplyVisuals(PlanetAPI planet) {
133                if (planet == null) return;
134                planet.getSpec().setShieldTexture2(null);
135                planet.getSpec().setShieldThickness2(0f);
136                planet.getSpec().setShieldColor2(null);
137                planet.applySpecChanges();
138                shownPlasmaNetVisuals = false;
139        }
140
141        protected boolean shownPlasmaNetVisuals = false;
142        
143        @Override
144        public void setSpecialItem(SpecialItemData special) {
145                super.setSpecialItem(special);
146
147                if (shownPlasmaNetVisuals && (special == null || !special.getId().equals(Items.PLASMA_DYNAMO))) {
148                        unapplyVisuals(market.getPlanetEntity());
149                }
150                
151                if (special != null && special.getId().equals(Items.PLASMA_DYNAMO)) {
152                        applyVisuals(market.getPlanetEntity());
153                }
154        }
155        
156        
157        
158}
