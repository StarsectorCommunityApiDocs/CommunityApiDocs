001package com.fs.starfarer.api.impl.campaign.econ;
002
003import com.fs.starfarer.api.campaign.econ.MarketAPI;
004import com.fs.starfarer.api.campaign.econ.MarketConditionAPI;
005import com.fs.starfarer.api.impl.campaign.ids.Conditions;
006import com.fs.starfarer.api.impl.campaign.intel.BaseIntelPlugin;
007import com.fs.starfarer.api.ui.TooltipMakerAPI;
008import com.fs.starfarer.api.util.Misc;
009import com.fs.starfarer.api.util.TimeoutTracker;
010
011public class RecentUnrest extends BaseMarketConditionPlugin {
012        
013        public static RecentUnrest get(MarketAPI market) {
014                return get(market, true);
015        }
016        public static RecentUnrest get(MarketAPI market, boolean addIfNeeded) {
017                MarketConditionAPI mc = market.getCondition(Conditions.RECENT_UNREST);
018                if (mc == null && !addIfNeeded) return null;
019                
020                if (mc == null) {
021                        String id = market.addCondition(Conditions.RECENT_UNREST);
022                        mc = market.getSpecificCondition(id);
023                }
024                return (RecentUnrest) mc.getPlugin();
025        }
026        
027        public static int getPenalty(MarketAPI market) {
028                RecentUnrest ru = get(market, false);
029                if (ru == null) return 0;
030                return ru.getPenalty();
031        }
032        
033        public static float DECREASE_DAYS = 90f;
034        
035        protected int penalty;
036        protected float untilDecrease = DECREASE_DAYS;
037        protected TimeoutTracker<String> reasons = new TimeoutTracker<String>();
038        
039        public RecentUnrest() {
040        }
041
042        public int getPenalty() {
043                return penalty;
044        }
045
046        public void apply(String id) {
047                market.getStability().modifyFlat(id, -1 * penalty, "Recent unrest");
048        }
049
050        public void unapply(String id) {
051                market.getStability().unmodify(id);
052        }
053        
054        
055        public void add(int stability, String reason) {
056                penalty += stability;
057                float dur = reasons.getRemaining(reason) + stability * DECREASE_DAYS;
058                reasons.add(reason, dur);
059        }
060        
061        public void counter(int points, String reason) {
062                points = Math.min(points, penalty);
063                penalty -= points;
064                if (penalty < 0) penalty = 0;
065                float dur = reasons.getRemaining(reason) + points * DECREASE_DAYS;
066                reasons.add(reason, dur);
067        }
068        
069        @Override
070        public void advance(float amount) {
071                super.advance(amount);
072                
073                float days = Misc.getDays(amount);
074                //days *= 100000f;
075                reasons.advance(days);
076
077                if (penalty > 0) {
078                        untilDecrease -= days;
079                        if (untilDecrease <= 0) {
080                                penalty--;
081                                if (penalty < 0) penalty = 0;
082                                untilDecrease = DECREASE_DAYS;
083                        }
084                }
085                if (penalty <= 0) {
086                        market.removeSpecificCondition(condition.getIdForPluginModifications());
087                }
088        }
089
090        @Override
091        public void createTooltip(TooltipMakerAPI tooltip, boolean expanded) {
092                super.createTooltip(tooltip, expanded);
093        }
094        
095
096        @Override
097        protected void createTooltipAfterDescription(TooltipMakerAPI tooltip, boolean expanded) {
098                super.createTooltipAfterDescription(tooltip, expanded);
099                
100                float pad = 3f;
101                float opad = 10f;
102                
103                tooltip.addPara("%s stability. Goes down by one point every three months.", 
104                                opad, Misc.getHighlightColor(),
105                                "-" + (int)penalty);
106                
107                if (!reasons.getItems().isEmpty()) {
108                        tooltip.addPara("Recent contributing factors:", opad);
109                        
110                        float initPad = pad;
111                        for (String reason : reasons.getItems()) {
112                                tooltip.addPara(BaseIntelPlugin.BULLET + reason, initPad);
113                                initPad = 0f;
114                        }
115                }
116                
117        }
118
119        @Override
120        public boolean isTransient() {
121                return false;
122        }
123        
124        public void setPenalty(int penalty) {
125                this.penalty = penalty;
126        }
127        
128        
129}
