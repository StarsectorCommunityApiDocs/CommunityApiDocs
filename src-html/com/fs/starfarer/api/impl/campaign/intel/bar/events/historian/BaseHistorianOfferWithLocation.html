029public abstract class BaseHistorianOfferWithLocation extends BaseHistorianOffer implements ExtraSalvageShownListener {
030
031        private SectorEntityToken entity;
032        
033        public BaseHistorianOfferWithLocation(SectorEntityToken entity) {
034                super();
035                this.entity = entity;
036                setImportant(true);
037        }
038
039        @Override
040        public void init(InteractionDialogAPI dialog) {
041                super.init(dialog);
042                
043                setDone(true);
044                setRemove(true);
045                setEndConversationOnReturning(true);
046                
047                Global.getSector().getIntelManager().addIntel(this, false, text);
048                Global.getSector().getListenerManager().addListener(this);
049                
050                Misc.makeImportant(entity, getClass().getSimpleName());
051                
052                CargoAPI cargo = Global.getFactory().createCargo(true);
053                addItemToCargo(cargo);
054                BaseSalvageSpecial.addExtraSalvage(cargo, entity.getMemoryWithoutUpdate(), -1);
055        }
056        
057
058        public void reportExtraSalvageShown(SectorEntityToken entity) {
059                if (this.entity != entity) return;
060                
061                Misc.makeUnimportant(entity, getClass().getSimpleName());
062                endIntel();
063        }
064        
065        protected void endIntel() {
066                if (isEnding() || isEnded()) return;
067                
068                endAfterDelay();
069                Global.getSector().addScript(this);
070                Global.getSector().getListenerManager().removeListener(this);
071                
072                if (Misc.random.nextFloat() < 0.5f) return;
073                if (this instanceof WeaponBlueprintOffer) return;
074                if (this instanceof FighterBlueprintOffer) return;
075                
076                DelayedFleetEncounter e = new DelayedFleetEncounter(null, "hist");
077                e.setDelayNone();
078                //e.setEncounterInHyper();
079                e.setLocationAnywhere(true, Factions.LUDDIC_PATH);
080                e.beginCreate();
081                e.triggerCreateFleet(FleetSize.LARGE, FleetQuality.DEFAULT, Factions.LUDDIC_PATH, FleetTypes.PATROL_LARGE, new Vector2f());
082                
083                float q = 0f;
084                if (this instanceof ShipBlueprintOffer) {
085                        ShipHullSpecAPI spec = Global.getSettings().getHullSpec(((ShipBlueprintOffer)this).data);
086                        if (spec != null) {
087                                q = Math.min(1f, (float)spec.getFleetPoints() / 20f);
088                        }
089                } else if (this instanceof SpecialItemOffer) {
090                        q = 1f;
091                }
092//              else if (this instanceof WeaponBlueprintOffer) {
093//                      WeaponSpecAPI spec = Global.getSettings().getWeaponSpec(((WeaponBlueprintOffer)this).data);
094//                      if (spec != null) {
095//                              q = Math.min(0.5f, (float)spec.getOrdnancePointCost(null) / 40f);
096//                      }
097//              } else if (this instanceof FighterBlueprintOffer) {
098//                      FighterWingSpecAPI spec = Global.getSettings().getFighterWingSpec(((FighterBlueprintOffer)this).data);
099//                      if (spec != null) {
100//                              q = Math.min(0.5f, (float)spec.getOpCost(null) / 40f);
101//                      }
102//              }
103                
104                e.triggerSetAdjustStrengthBasedOnQuality(true, q);
105                e.triggerSetStandardAggroInterceptFlags();
106                e.triggerSetFleetGenericHailPermanent("HistorianPatherHail");
107                e.triggerFleetPatherNoDefaultTithe();
108                e.endCreate();
109        }
110        
111        
112        protected abstract void addItemToCargo(CargoAPI loot);
113        public abstract String getName();
114        
115
116        protected void addBulletPoints(TooltipMakerAPI info, ListInfoMode mode) {
117        }
118        
119        
120        @Override
121        public void createIntelInfo(TooltipMakerAPI info, ListInfoMode mode) {
122                if (!entity.isAlive()) {
123                        endIntel();
124                }
125
126                Color c = getTitleColor(mode);
127                info.addPara(getName(), c, 0f);
128                
129                addBulletPoints(info, mode);
130        }
131        
132        
133        public String getSmallDescriptionTitle() {
134                return getName();
135        }
136        
137
138        @Override
139        public void createSmallDescription(TooltipMakerAPI info, float width, float height) {
140                Color h = Misc.getHighlightColor();
141                Color g = Misc.getGrayColor();
142                //Color c = getTitleColor(mode);
143                Color tc = Misc.getTextColor();
144                float pad = 3f;
145                float opad = 10f;
146                
147                HistorianData hd = HistorianData.getInstance();
148                
149                info.addPara("The historian, " + hd.getPerson().getNameString() + ", has given you information about " +
150                                "the location of a valuable item.", opad);
151                
152                info.addSectionHeading("Item", Alignment.MID, opad);
153                
154                CargoAPI cargo = Global.getFactory().createCargo(true);
155                addItemToCargo(cargo);
156                info.showCargo(cargo, 10, true, opad);
157                
158                info.addSectionHeading("Location", Alignment.MID, opad);
159                if (isEnding()) {
160                        info.addPara("You have recovered this item.", opad);
161                } else {
162                        String located = BreadcrumbSpecial.getLocatedString(entity, true);
163                        if (entity instanceof PlanetAPI) {
164                                PlanetAPI planet = (PlanetAPI) entity;
165                                info.addPara("The item is in the ruins found on a " + 
166                                                                planet.getTypeNameWithLowerCaseWorld().toLowerCase() + " " + located + ".", opad);
167                                
168                                info.addPara("A full survey will need to be run before the ruins can be explored.", opad);
169                                
170                                info.showFullSurveyReqs(planet, true, opad);
171                                
172                        } else if (entity.getCustomEntitySpec() != null) {
173                                info.addPara("The item is inside " + entity.getCustomEntitySpec().getAOrAn() + " " + 
174                                                         entity.getCustomEntitySpec().getNameInText() + " " + located + ".", opad);
175                        } else {
176                                info.addPara("The item inside a " + entity.getName() + " " + located + ".", opad);
177                        }
178                }
179                
180
181        }
182        
183        
184        public Set<String> getIntelTags(SectorMapAPI map) {
185                Set<String> tags = super.getIntelTags(map);
186                tags.add(Tags.INTEL_EXPLORATION);
187                tags.add(Tags.INTEL_ACCEPTED);
188                return tags;
189        }
190
191        @Override
192        public SectorEntityToken getMapLocation(SectorMapAPI map) {
193                if (entity != null && entity.isDiscoverable() && entity.getStarSystem() != null) {
194                        return entity.getStarSystem().getCenter();
195                }
196                return entity;
197        }
198
199}
