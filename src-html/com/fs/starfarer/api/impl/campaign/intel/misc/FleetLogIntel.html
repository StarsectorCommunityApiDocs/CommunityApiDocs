003import java.util.Set;
004
005import com.fs.starfarer.api.Global;
006import com.fs.starfarer.api.campaign.CampaignTerrainAPI;
007import com.fs.starfarer.api.campaign.PlanetAPI;
008import com.fs.starfarer.api.campaign.SectorEntityToken;

010import com.fs.starfarer.api.impl.campaign.intel.BaseIntelPlugin;

014/**
015 * Implements EveryFrameScript because of deriving from BaseIntelPlugin, but not actually
016 * expected to be added as a script to anything unless that's necessary for the specific subclass.
017 *
018 * Copyright 2018 Fractal Softworks, LLC
019 */
020public class FleetLogIntel extends BaseIntelPlugin {
021
022        public static float DEFAULT_DURATION = 365f;
023
024        protected Float duration = null; 
025        protected SectorEntityToken removeTrigger = null;
026        protected String icon = null;
027        protected String sound = null;
028        
029        public FleetLogIntel() {
030        }
031        
032        public FleetLogIntel(float duration) {
033                this.duration = duration;
034        }
035
036        public void setDefaultExpiration() {
037                duration = DEFAULT_DURATION;
038        }
039        
040        public void setDuration(float days) {
041                duration = days;
042        }
043
044        @Override
045        public boolean shouldRemoveIntel() {
046                if (isEnded()) return true;
047                
048                if (removeTrigger != null) {
049                        if (!removeTrigger.isAlive()) return true;
050                        if (removeTrigger instanceof CampaignTerrainAPI) {
051                                CampaignTerrainAPI terrain = (CampaignTerrainAPI) removeTrigger;
052                                if (terrain.getPlugin() instanceof DebrisFieldTerrainPlugin) {
053                                        DebrisFieldTerrainPlugin debris = (DebrisFieldTerrainPlugin) terrain.getPlugin();
054                                        if (debris.isScavenged()) return true;
055                                }
056                        } else if (removeTrigger instanceof PlanetAPI) {
057                                
058                        }
059                }
060                
061                if (isImportant() || duration == null) return false;
062
063                Long ts = getPlayerVisibleTimestamp();
064                if (ts == null) return false;
065                return Global.getSector().getClock().getElapsedDaysSince(ts) >= duration;
066                
067//              float dur = DEFAULT_DURATION;
068//              if (duration != null) dur = duration;
069//              return Global.getSector().getClock().getElapsedDaysSince(ts) >= dur;
070        }
071
072        
073        public void setIcon(String icon) {
074                this.icon = icon;
075        }
076
077        @Override
078        public String getIcon() {
079                if (icon != null) return icon;
080                return Global.getSettings().getSpriteName("intel", "fleet_log");
081        }
082
083        @Override
084        public Set<String> getIntelTags(SectorMapAPI map) {
085                //Set<String> tags = new LinkedHashSet<String>(); ??? why probably bug
086                Set<String> tags = super.getIntelTags(map);
087                tags.add(Tags.INTEL_FLEET_LOG);
088                return tags;
089        }
090
091        public SectorEntityToken getRemoveTrigger() {
092                return removeTrigger;
093        }
094
095        public void setRemoveTrigger(SectorEntityToken removeTrigger) {
096                this.removeTrigger = removeTrigger;
097        }
098        
099        public String getSound() {
100                return sound;
101        }
102
103        public void setSound(String sound) {
104                this.sound = sound;
105        }
106        
107        @Override
108        public String getCommMessageSound() {
109                if (sound != null) return sound;
110                return getSoundLogUpdate();
111        }
112        
113}
