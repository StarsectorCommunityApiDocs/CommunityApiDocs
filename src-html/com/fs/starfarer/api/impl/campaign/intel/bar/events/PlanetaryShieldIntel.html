028        public static enum PSIStage {
029                TALK_TO_PILOT,
030                GO_TO_PLANET,
031                DONE,
032                ;
033        }
034        
035        public static int FINISHED_XP = 20000;
036        public static int PAY_PILOT_XP = 5000;
037        
038        protected PlanetAPI planet;
039        protected PlanetaryShieldBarEvent event;
040        
041        protected PSIStage stage;
042        protected int pilotCredits;
043        
044        public PlanetaryShieldIntel(PlanetAPI planet, PlanetaryShieldBarEvent event) {
045                this.planet = planet;
046                this.event = event;
047                
048                PersonAPI pilot = event.getPilot();
049                Misc.makeImportant(pilot, "psi");
050                MarketAPI market = event.getPilotMarket();
051                market.addPerson(pilot);
052                market.getCommDirectory().addPerson(pilot);
053                
054                pilotCredits = 10000 + 1000 * Misc.random.nextInt(10);
055                
056                pilot.getMemoryWithoutUpdate().set("$psi_isPilot", true);
057                pilot.getMemoryWithoutUpdate().set("$psi_eventRef", this);
058                pilot.getMemoryWithoutUpdate().set("$psi_credits", Misc.getDGSCredits(pilotCredits));
059                
060                //Misc.makeImportant(planet, "saci");
061                //cache.getMemoryWithoutUpdate().set("$saic_eventRef", this);
062                //Global.getSector().addScript(this);
063                
064                stage = PSIStage.TALK_TO_PILOT;
065        }
066        
067        @Override
068        protected void notifyEnded() {
069                super.notifyEnded();
070                Global.getSector().removeScript(this);
071                
072                PersonAPI pilot = event.getPilot();
073                MarketAPI market = event.getPilotMarket();
074                market.removePerson(pilot);
075                market.getCommDirectory().removePerson(pilot);
076                Misc.makeUnimportant(planet, "psi");
077        }

081        @Override
082        public boolean callEvent(String ruleId, InteractionDialogAPI dialog,
083                                                         List<Token> params, Map<String, MemoryAPI> memoryMap) {
084                String action = params.get(0).getString(memoryMap);
085                
086                CampaignFleetAPI playerFleet = Global.getSector().getPlayerFleet();
087                CargoAPI cargo = playerFleet.getCargo();
088                //MemoryAPI memory = planet.getMemoryWithoutUpdate();
089                
090                PersonAPI pilot = event.getPilot();
091                MarketAPI market = event.getPilotMarket();
092                
093                if (action.equals("prepare")) {
094                        pilot.getMemoryWithoutUpdate().set("$psi_credits", Misc.getDGSCredits(pilotCredits), 0);
095                        pilot.getMemoryWithoutUpdate().set("$psi_playerCredits", Misc.getDGSCredits(cargo.getCredits().get()), 0);
096                } else if (action.equals("canPay")) {
097                        return cargo.getCredits().get() >= pilotCredits;
098                } else if (action.equals("payPilot")) {
099                        market.removePerson(pilot);
100                        market.getCommDirectory().removePerson(pilot);
101                        
102                        cargo.getCredits().subtract(pilotCredits);
103                        AddRemoveCommodity.addCreditsLossText(pilotCredits, dialog.getTextPanel());
104                        Global.getSector().getPlayerPerson().getStats().addXP(PAY_PILOT_XP, dialog.getTextPanel());
105
106                        Misc.makeImportant(planet, "psi");
107                        stage = PSIStage.GO_TO_PLANET;
108                        sendUpdate(PSIStage.GO_TO_PLANET, dialog.getTextPanel());
109                }
110                
111                return true;
112        }

127        protected void addBulletPoints(TooltipMakerAPI info, ListInfoMode mode) {
128                
129                Color h = Misc.getHighlightColor();
130                Color g = Misc.getGrayColor();
131                float pad = 3f;
132                float opad = 10f;
133                
134                float initPad = pad;
135                if (mode == ListInfoMode.IN_DESC) initPad = opad;
136                
137                Color tc = getBulletColorForMode(mode);
138                
139                bullet(info);
140                boolean isUpdate = getListInfoParam() != null;
141                
142                MarketAPI market = event.getPilotMarket();
143                
144                if (stage == PSIStage.TALK_TO_PILOT) {
145                        info.addPara("Talk to the pilot at %s", initPad, tc, market.getFaction().getBaseUIColor(), market.getName());
146                } else if (stage == PSIStage.GO_TO_PLANET) {
147                        info.addPara("Explore the planet", tc, initPad);
148                }
149                
150                initPad = 0f;
151                
152                unindent(info);
153        }

156        @Override
157        public void createIntelInfo(TooltipMakerAPI info, ListInfoMode mode) {
158                Color c = getTitleColor(mode);
159                info.setParaSmallInsignia();
160                info.addPara(getName(), c, 0f);
161                info.setParaFontDefault();
162                addBulletPoints(info, mode);
163                
164        }
165        
166        @Override
167        public void createSmallDescription(TooltipMakerAPI info, float width, float height) {
168                Color h = Misc.getHighlightColor();
169                Color g = Misc.getGrayColor();
170                Color tc = Misc.getTextColor();
171                float pad = 3f;
172                float opad = 10f;
173
174                if (stage == PSIStage.TALK_TO_PILOT) {
175                        info.addPara("An old spacer told you a tale about a mysterious red planet. " +
176                                                 "The pilot - the only other survivor of the salvage expedition - may know the planet's location.", opad);
177                } else if (stage == PSIStage.GO_TO_PLANET) {
178                        info.addPara("You've talked to the old spacer's pilot comrade and convinced " +
179                                                 "them to divulge the location of the planet.", opad);
180                } else {
181                        info.addPara("You've found the planet and uncovered its secret.", opad);
182                }
183
184                addBulletPoints(info, ListInfoMode.IN_DESC);
185                
186        }
