001package com.fs.starfarer.api.impl.campaign.intel.group;
002
003import com.fs.starfarer.api.Global;
004import com.fs.starfarer.api.campaign.CampaignFleetAPI;
005import com.fs.starfarer.api.impl.campaign.ids.FleetTypes;
006import com.fs.starfarer.api.impl.campaign.ids.MemFlags;
007import com.fs.starfarer.api.impl.campaign.missions.FleetCreatorMission;
008import com.fs.starfarer.api.util.IntervalUtil;
009import com.fs.starfarer.api.util.Misc;
010
011
012
013public class SindrianDiktatPunitiveExpedition extends GenericRaidFGI  {
014
015        public static final String SDPE_FLEET = "$SDPE_fleet";
016        
017        public static String KEY = "$SDPE_ref";
018        public static SindrianDiktatPunitiveExpedition get() {
019                return (SindrianDiktatPunitiveExpedition) Global.getSector().getMemoryWithoutUpdate().get(KEY);
020        }
021        
022        
023        protected IntervalUtil interval = new IntervalUtil(0.1f, 0.3f);
024        
025        
026        public SindrianDiktatPunitiveExpedition(GenericRaidParams params) {
027                super(params);
028                
029                Global.getSector().getMemoryWithoutUpdate().set(KEY, this);
030        }
031        
032        @Override
033        protected void notifyEnding() {
034                super.notifyEnding();
035                
036                Global.getSector().getMemoryWithoutUpdate().unset(KEY);
037        }
038        
039        @Override
040        protected void notifyEnded() {
041                super.notifyEnded();
042        }
043        
044
045        @Override
046        public String getNoun() {
047                return super.getNoun();
048                //return "punitive expedition";
049        }
050
051        @Override
052        public String getForcesNoun() {
053                return super.getForcesNoun();
054        }
055
056
057        @Override
058        public String getBaseName() {
059                return super.getBaseName();
060                //return Misc.ucFirst(getFaction().getPersonNamePrefix()) + " " + "Punitive Expedition";
061        }
062
063
064        @Override
065        protected void preConfigureFleet(int size, FleetCreatorMission m) {
066                m.setFleetTypeMedium(FleetTypes.TASK_FORCE); // default would be "Patrol", don't want that
067        }
068        
069        @Override
070        protected void configureFleet(int size, FleetCreatorMission m) {
071                m.triggerSetFleetFlag(SDPE_FLEET);
072                if (size >= 8) {
073                        m.triggerSetFleetDoctrineOther(5, 0); // more capitals in large fleets
074                }
075        }
076
077        
078        @Override
079        public void abort() {
080                if (!isAborted()) {
081                        for (CampaignFleetAPI curr : getFleets()) {
082                                curr.getMemoryWithoutUpdate().unset(SDPE_FLEET);
083                        }
084                }
085                super.abort();
086        }
087        
088        
089
090        @Override
091        public void advance(float amount) {
092                super.advance(amount);
093                
094                float days = Misc.getDays(amount);
095                interval.advance(days);
096                
097                if (interval.intervalElapsed()) {
098                        if (isCurrent(PAYLOAD_ACTION)) {
099                                String reason = "SDPunEx";
100                                for (CampaignFleetAPI curr : getFleets()) {
101                                        Misc.setFlagWithReason(curr.getMemoryWithoutUpdate(), MemFlags.MEMORY_KEY_MAKE_HOSTILE,
102                                                                                        reason, true, 1f);
103                                }
104                                
105                        }
106                }
107        }
108
109
110        
111}
