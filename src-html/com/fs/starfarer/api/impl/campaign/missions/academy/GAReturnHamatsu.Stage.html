025
026public class GAReturnHamatsu extends GABaseMission { //implements ShipRecoveryListener {
027
028        public static enum Stage {
029                RETURN_HAMATSU,
030                COMPLETED,
031                FAILED,
032        }
033        
034        protected PersonAPI callisto;
035        
036        @Override
037        protected boolean create(MarketAPI createdAt, boolean barEvent) {
038                if (!setGlobalReference("$gaRH_ref", null)) {
039                        return false;
040                }
041                
042                callisto = getImportantPerson(People.IBRAHIM);
043                if (callisto == null) return false;
044                
045                setPersonOverride(callisto);
046                
047                //setStoryMission();
048                
049                setStartingStage(Stage.RETURN_HAMATSU);
050                addSuccessStages(Stage.COMPLETED);
051                addFailureStages(Stage.FAILED);
052                
053                connectWithGlobalFlag(Stage.RETURN_HAMATSU, Stage.COMPLETED, "$gaRH_completed");
054                connectWithGlobalFlag(Stage.RETURN_HAMATSU, Stage.FAILED, "$gaRH_failed");
055                makeImportant(callisto, "$gaRH_returnHere", Stage.RETURN_HAMATSU);
056                
057                setCreditReward(100000);
058                setRepRewardPerson(RepRewards.EXTREME);
059                setRepRewardFaction(RepRewards.HIGH);
060                
061                setRepPenaltyPerson(0f); // handled in rules, is in fact very high
062                setRepPenaltyFaction(0f);
063                
064                setPersonIsPotentialContactOnSuccess(callisto, 1f);
065                
066                beginStageTrigger(Stage.COMPLETED);
067                triggerSetGlobalMemoryValue("$gaRH_missionCompleted", true);
068                triggerMakeNonStoryCritical(callisto.getMarket());
069                endTrigger();
070                
071                return true;
072        }
073        
074        protected boolean callAction(String action, String ruleId, InteractionDialogAPI dialog, List<Token> params,
075                                                                 Map<String, MemoryAPI> memoryMap) {
076
077                if ("refusedToReturn".equals(action)) {
078                        if (rollProbability(0.5f)) {
079                                DelayedFleetEncounter e = new DelayedFleetEncounter(genRandom, getMissionId());
080                                e.setDelayMedium();
081                                e.setLocationInnerSector(false, Factions.INDEPENDENT);
082                                e.beginCreate();
083                                e.triggerCreateFleet(FleetSize.MEDIUM, FleetQuality.VERY_HIGH, Factions.MERCENARY, FleetTypes.PATROL_LARGE, new Vector2f());
084                                e.triggerSetFleetOfficers(OfficerNum.MORE, OfficerQuality.HIGHER);
085                                e.triggerFleetSetFaction(Factions.INDEPENDENT);
086                                e.triggerSetFleetFlag("$gaRH_consequences");
087                                e.triggerMakeNoRepImpact();
088                                e.triggerSetStandardAggroInterceptFlags();
089                                e.endCreate();
090                        }
091                        return true;
092                } else if ("transferHamatsu".equals(action)) {
093                        FleetMemberAPI hamatsu = null;
094                        for (FleetMemberAPI member : Global.getSector().getPlayerFleet().getFleetData().getMembersListCopy()) {
095                                if (member.getId().equals(TTBlackSite.HAMATSU_ID)) {
096                                        hamatsu = member;
097                                        break;
098                                }
099                        }
100                        if (hamatsu != null) {
101                                AddRemoveCommodity.addFleetMemberLossText(hamatsu, dialog.getTextPanel());
102                                Global.getSector().getPlayerFleet().getFleetData().removeFleetMember(hamatsu);
103                        }
104                        
105                        return true;
106                }
107                return false;
108        }
109        
110        protected void updateInteractionDataImpl() {
111                set("$gaRH_stage", getCurrentStage());
112                set("$gaRH_reward", Misc.getWithDGS(getCreditsReward()));
113        }
114        
115        @Override
116        public void addDescriptionForNonEndStage(TooltipMakerAPI info, float width, float height) {
117                float opad = 10f;
118                Color h = Misc.getHighlightColor();
119                if (currentStage == Stage.RETURN_HAMATSU) {
120                        addStandardMarketDesc("Deliver the ISS Hamatsu to " + callisto.getNameString()
121                                        + " " + callisto.getMarket().getOnOrAt() + "", callisto.getMarket(), info, opad);
122                        info.addPara("She is sentimental about the ship - her first independent command - and is willing to pay a "
123                                        + "large sum for its return.", opad);
124                }
125        }
126
127        @Override
128        public boolean addNextStepText(TooltipMakerAPI info, Color tc, float pad) {
129                Color h = Misc.getHighlightColor();
130                if (currentStage == Stage.RETURN_HAMATSU) {
131                        info.addPara("Deliver the ISS Hamatsu to " + callisto.getNameString(), tc, pad);
132                        return true;
133                }
134                return false;
135        }
136
137        @Override
138        public String getBaseName() {
139                return "Return the ISS Hamatsu";
140        }
141
142        @Override
143        public String getPostfixForState() {
144                if (startingStage != null) {
145                        return "";
146                }
147                return super.getPostfixForState();
148        }
149        
150        
151}
