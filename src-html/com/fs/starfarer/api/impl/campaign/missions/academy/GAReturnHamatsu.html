026public class GAReturnHamatsu extends GABaseMission { //implements ShipRecoveryListener {
027
028        public static enum Stage {
029                RETURN_HAMATSU,
030                COMPLETED,
031                FAILED,
032        }
033        
034        protected PersonAPI callisto;
035        
036        @Override
037        protected boolean create(MarketAPI createdAt, boolean barEvent) {
038                if (!setGlobalReference("$gaRH_ref", null)) {
039                        return false;
040                }
041                
042                callisto = getImportantPerson(People.IBRAHIM);
043                if (callisto == null) return false;
044                
045                setPersonOverride(callisto);
046                
047                //setStoryMission();
048                
049                setStartingStage(Stage.RETURN_HAMATSU);
050                addSuccessStages(Stage.COMPLETED);
051                addFailureStages(Stage.FAILED);
052                
053                connectWithGlobalFlag(Stage.RETURN_HAMATSU, Stage.COMPLETED, "$gaRH_completed");
054                connectWithGlobalFlag(Stage.RETURN_HAMATSU, Stage.FAILED, "$gaRH_failed");
055                makeImportant(callisto, "$gaRH_returnHere", Stage.RETURN_HAMATSU);
056                
057                setCreditReward(100000);
058                setRepRewardPerson(RepRewards.EXTREME);
059                setRepRewardFaction(RepRewards.HIGH);
060                
061                setRepPenaltyPerson(0f); // handled in rules, is in fact very high
062                setRepPenaltyFaction(0f);
063                
064                setPersonIsPotentialContactOnSuccess(callisto, 1f);
065                
066                return true;
067        }
068        
069        protected boolean callAction(String action, String ruleId, InteractionDialogAPI dialog, List<Token> params,
070                                                                 Map<String, MemoryAPI> memoryMap) {
071
072                if ("refusedToReturn".equals(action)) {
073                        if (rollProbability(0.5f)) {
074                                DelayedFleetEncounter e = new DelayedFleetEncounter(genRandom, getMissionId());
075                                e.setDelayMedium();
076                                e.setLocationInnerSector(false, Factions.INDEPENDENT);
077                                e.beginCreate();
078                                e.triggerCreateFleet(FleetSize.MEDIUM, FleetQuality.VERY_HIGH, Factions.MERCENARY, FleetTypes.PATROL_LARGE, new Vector2f());
079                                e.triggerSetFleetOfficers(OfficerNum.MORE, OfficerQuality.HIGHER);
080                                e.triggerFleetSetFaction(Factions.INDEPENDENT);
081                                e.triggerSetFleetFlag("$gaRH_consequences");
082                                e.triggerMakeNoRepImpact();
083                                e.triggerSetStandardAggroInterceptFlags();
084                                e.endCreate();
085                        }
086                        return true;
087                } else if ("transferHamatsu".equals(action)) {
088                        FleetMemberAPI hamatsu = null;
089                        for (FleetMemberAPI member : Global.getSector().getPlayerFleet().getFleetData().getMembersListCopy()) {
090                                if (member.getId().equals(TTBlackSite.HAMATSU_ID)) {
091                                        hamatsu = member;
092                                        break;
093                                }
094                        }
095                        if (hamatsu != null) {
096                                AddRemoveCommodity.addFleetMemberLossText(hamatsu, dialog.getTextPanel());
097                                Global.getSector().getPlayerFleet().getFleetData().removeFleetMember(hamatsu);
098                        }
099                        
100                        return true;
101                }
102                return false;
103        }
104        
105        protected void updateInteractionDataImpl() {
106                set("$gaRH_stage", getCurrentStage());
107                set("$gaRH_reward", Misc.getWithDGS(getCreditsReward()));
108        }
109        
110        @Override
111        public void addDescriptionForNonEndStage(TooltipMakerAPI info, float width, float height) {
112                float opad = 10f;
113                Color h = Misc.getHighlightColor();
114                if (currentStage == Stage.RETURN_HAMATSU) {
115                        addStandardMarketDesc("Deliver the ISS Hamatsu to " + callisto.getNameString()
116                                        + " " + callisto.getMarket().getOnOrAt() + "", callisto.getMarket(), info, opad);
117                        info.addPara("She is sentimental about the ship - her first independent command - and is willing to pay a "
118                                        + "large sum for its return.", opad);
119                }
120        }
121
122        @Override
123        public boolean addNextStepText(TooltipMakerAPI info, Color tc, float pad) {
124                Color h = Misc.getHighlightColor();
125                if (currentStage == Stage.RETURN_HAMATSU) {
126                        info.addPara("Deliver the ISS Hamatsu to " + callisto.getNameString(), tc, pad);
127                        return true;
128                }
129                return false;
130        }
131
132        @Override
133        public String getBaseName() {
134                return "Return the ISS Hamatsu";
135        }
136
137        @Override
138        public String getPostfixForState() {
139                if (startingStage != null) {
140                        return "";
141                }
142                return super.getPostfixForState();
143        }
144        
145        
146}
