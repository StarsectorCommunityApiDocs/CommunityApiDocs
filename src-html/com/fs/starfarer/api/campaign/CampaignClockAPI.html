001package com.fs.starfarer.api.campaign;
002
003import java.util.GregorianCalendar;
004
005
006/**
007 * @author Alex Mosolov
008 *
009 * Copyright 2012 Fractal Softworks, LLC
010 */
011public interface CampaignClockAPI {
012        public int getCycle();
013        /**
014         * 1 = January, 12 = December.
015         * @return
016         */
017        public int getMonth();
018
019        /**
020         * Get the day since game start. If game started 4 in-game days ago, it will return 4.0.
021         * Example: Global.getSector().getClock().getDay()
022         * @return int value
023         */
024        public int getDay();
025        public int getHour();
026        public float convertToDays(float realSeconds);
027        public float convertToMonths(float realSeconds);
028        /**
029         * Gets the timestamp of the current date as a 14 digit, negative signed long number. March 2, c206 is -55661253120000.
030         * Example: Global.getSector().getClock().getTimestamp()
031         * @return long value
032         */
033        public long getTimestamp();
034        /**
035         * Gets the time elasped after the specified timestamp.
036         * Example: Returns the time in float time. If partial day, it will return 0.5966898.
037         * @params long timestamp
038         * @return float value
039         */
040        public float getElapsedDaysSince(long timestamp);
041        
042        public String getMonthString();
043        public String getShortMonthString();
044        public float getSecondsPerDay();
045        
046        /**
047         * New clock based on the timestamp.
048         * @param timestamp
049         * @return
050         */
051        CampaignClockAPI createClock(long timestamp);
052        String getDateString();
053        float convertToSeconds(float days);
054        String getShortDate();
055        String getCycleString();
056        GregorianCalendar getCal();
057}
