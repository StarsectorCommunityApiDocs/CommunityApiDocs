028public class CoreDiscoverEntityPlugin implements DiscoverEntityPlugin, 
029                                        CurrentLocationChangedListener, SurveyPlanetListener, DetectedEntityListener, ShowLootListener {
030        
031        /**
032         * How long since discovering or salvaging something before SalvorsTallyIntel pops up.
033         * Point is to not pop it up if it's just going to go away real soon anyway.
034         */
035        public static float SALVORS_TALLY_DELAY_SECONDS = 10f;
036        
037        public void discoverEntity(SectorEntityToken entity) {
038                
039                entity.setDiscoverable(null);
040                entity.setSensorProfile(null);
041                
042                if (entity.hasTag(Tags.WARNING_BEACON)) {
043                        WarningBeaconIntel intel = new WarningBeaconIntel(entity);
044                        Global.getSector().getIntelManager().addIntel(intel);
045                } else {
046                        Color c = Global.getSector().getPlayerFaction().getBaseUIColor();
047                        MessageIntel intel = new MessageIntel("Discovered: " + entity.getName(),
048                                                                                                  c, new String[] {entity.getName()}, c);
049                        intel.setSound("ui_discovered_entity");
050                        intel.setIcon(Global.getSettings().getSpriteName("intel", "discovered_entity"));
051                        Global.getSector().getCampaignUI().addMessage(intel);
052                }
053                
054
055                
056                float xp = 0;
057                if (entity.hasDiscoveryXP()) {
058                        xp = entity.getDiscoveryXP();
059                } else if (entity.getCustomEntityType() != null) {
060                        SalvageEntityGenDataSpec salvageSpec = (SalvageEntityGenDataSpec) Global.getSettings().getSpec(SalvageEntityGenDataSpec.class, entity.getCustomEntityType(), true);
061                        if (salvageSpec != null) {
062                                xp = salvageSpec.getXpDiscover();
063                        }
064                }
065                if (xp > 0) {
066                        Global.getSector().getPlayerPerson().getStats().addXP((long) xp);
067                }
068                
069                ListenerUtil.reportEntityDiscovered(entity);
070                
071                if (entity.hasTag(Tags.SALVAGEABLE)) {
072                        addSalvorsTallyIfNeeded(entity.getStarSystem());
073                }
074        }
075        
076
077        public int getHandlingPriority(Object params) {
078                return 0;
079        }
080
081
082        @Override
083        public void reportCurrentLocationChanged(LocationAPI prev, LocationAPI curr) {
084                if (prev instanceof StarSystemAPI) {
085                        StarSystemAPI system = (StarSystemAPI) prev;
086                        removeSalvorsTallyIfNeeded(system);
087                }
088                if (curr instanceof StarSystemAPI) {
089                        StarSystemAPI system = (StarSystemAPI) curr;
090                        addSalvorsTallyIfNeeded(system);
091                }
092        }
093        
094        @Override
095        public void reportPlayerSurveyedPlanet(PlanetAPI planet) {
096                if (Misc.hasUnexploredRuins(planet.getMarket())) {
097                        addSalvorsTallyIfNeeded(planet.getStarSystem());
098                }
099        }
100        
101        @Override
102        public void reportAboutToShowLootToPlayer(CargoAPI loot, InteractionDialogAPI dialog) {
103                SectorEntityToken target = dialog.getInteractionTarget();
104                if (target == null) return;
105                if (!target.hasTag(Tags.SALVAGEABLE)) return;
106                if (target.getStarSystem() == null) return;
107                
108                addSalvorsTallyIfNeeded(target.getStarSystem());
109        }
110        
111        @Override
112        public void reportDetectedEntity(SectorEntityToken entity, VisibilityLevel level) {
113                //System.out.println("Detected entity: " + entity.getName() + " - " + level.name());
114                if (entity instanceof CampaignFleetAPI && level == VisibilityLevel.COMPOSITION_AND_FACTION_DETAILS) {
115                        CampaignFleetAPI fleet = (CampaignFleetAPI) entity;
116                        if (fleet.isStationMode() && fleet.getFaction().getId().equals(Factions.REMNANTS)) {
117                                addRemnantNexusIntelIfNeeded(fleet);
118                        }
119                        
120                }
121        }
122        
123        
124        public void addRemnantNexusIntelIfNeeded(CampaignFleetAPI nexus) {
125                if (RemnantNexusIntel.getNexusIntel(nexus) == null) {
126                        new RemnantNexusIntel(nexus); // adds the intel
127                }
128        }
129        
130        
131        public static class SalvorsTallyAdder implements EveryFrameScript {
132
133                protected float delay = 0f;
134                protected boolean done = false;
135                protected SalvorsTallyIntel intel;
136                
137                public SalvorsTallyAdder(SalvorsTallyIntel intel, float delay) {
138                        this.delay = delay;
139                        this.intel = intel;
140                }
141                
142                @Override
143                public boolean isDone() {
144                        return done;
145                }
146
147                @Override
148                public boolean runWhilePaused() {
149                        return false;
150                }
151
152                @Override
153                public void advance(float amount) {
154                        delay -= amount;
155                        if (delay <= 0) {
156                                done = true;
157                                SalvorsTally tally = intel.computeTally();
158                                if (tally.value != SalvageValue.NONE) {
159                                        Global.getSector().getIntelManager().addIntel(intel);
160                                }
161                        }
162                }
163                
164        }
165        
166        public static void addSalvorsTallyIfNeeded(StarSystemAPI system) {
167                if (system == null) return;
168                if (system.hasTag(Tags.THEME_CORE)) return;
169                if (system.getType() == StarSystemType.DEEP_SPACE) return;
170                if (TutorialMissionIntel.isTutorialInProgress()) return;
171                
172                SalvorsTallyIntel intel = SalvorsTallyIntel.getSalvorsTallyIntel(system);
173                if (intel == null) {
174                        // Delay adding so it doesn't necessarily pop up for every little thing you see and
175                        // immediately salvage
176                        for (EveryFrameScript curr : Global.getSector().getScripts()) {
177                                if (curr instanceof SalvorsTallyAdder) {
178                                        SalvorsTallyAdder adder = (SalvorsTallyAdder) curr;
179                                        if (adder.intel.getSystem() == system) {
180                                                adder.delay = SALVORS_TALLY_DELAY_SECONDS;
181                                                return;
182                                        }
183                                }
184                        }               
185                        
186                        intel = new SalvorsTallyIntel(system);
187                        SalvorsTally tally = intel.computeTally();
188                        if (tally.value != SalvageValue.NONE) {
189                                SalvorsTallyAdder adder = new SalvorsTallyAdder(intel, SALVORS_TALLY_DELAY_SECONDS);
190                                Global.getSector().addScript(adder);
191                                //Global.getSector().getIntelManager().addIntel(intel);
192                        }
193                }
194        }
195        
196        public void removeSalvorsTallyIfNeeded(StarSystemAPI system) {
197                if (system == null) return;
198                SalvorsTallyIntel intel = SalvorsTallyIntel.getSalvorsTallyIntel(system);
199                if (intel != null) {
200                        SalvorsTally tally = intel.computeTally();
201                        if (tally.value == SalvageValue.NONE) {
202                                intel.endImmediately();
203                        }
204                }
205        }
